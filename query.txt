SELECT Vendedor.nome ,Vendedor.id,
			COUNT(Venda.Vendedor_id) AS qtdVendas
			 FROM Vendedor
			JOIN Venda
			 ON Venda.Vendedor_id = Vendedor.id
			GROUP BY Venda.Vendedor_id



SELECT VENDEDOR.NOME, VENDEDOR.ID,
			COUNT (VENDA.VENDEDOR_ID) AS QTD_VENDAS,
			AVG(VALOR_VENDA) AS MEDIA,
			SUM(VALOR_VENDA) AS VALOR_TOTAL_VENDAS
			FROM VENDEDOR
			INNER JOIN VENDA
			ON VENDA.VENDEDOR_ID = VENDEDOR.ID
			WHERE DATA_VENDA > '2022-08-01' AND DATA_VENDA < '2022-08-15'
			GROUP BY VENDA.VENDEDOR_ID
